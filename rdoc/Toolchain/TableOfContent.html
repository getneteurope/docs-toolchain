def self.convert_nodes(entry, nodes = {})
  return nodes if entry.nil? || entry.size.zero?

  nodes[entry['id']] = entry.only(%w[parents label]) unless
    entry['level'] == -1
  entry['children'].each do |e|
    convert_nodes(e, nodes)
  end
  return nodes
end
